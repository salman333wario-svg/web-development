// Object Data Structure Project: Student Records const students = [ { id: 1, name: "Amina", marks: [78, 82, 90] }, { id: 2, name: "Brian", marks: [65, 70, 68] }, { id: 3, name: "Salman", marks: [88, 91, 85] } ]; // Function to calculate average function getAverage(marksArray) { let total = 0; for (let mark of marksArray) { total += mark; } return total / marksArray.length; } // Display student data using objects and arrays for (let student of students) { console.log("ID:", student.id); console.log("Name:", student.name); console.log("Average:", getAverage(student.marks)); console.log("-------------------"); }
